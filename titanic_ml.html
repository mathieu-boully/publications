def split(df,p=0.6): 
    # fonction split qui spare et retourne deux DataFrames selon la valeur de p (=train_size)
    # entres : 
        # - df : DataFrame complet
        # - p : taille de train_size (par dfaut p=0.6)
    # sorties : 
        # - train : chantillon d'entranement
        # - test : : chantillon de test
    train = df.sample(frac=p,replace=False)
    test = df.sample(frac=(1-p),replace=False)
    return train, test

from math import *

def log2(x):
    # fonction pour calculer le log2 utilis dans la fonction entropie
    return log(x)/log(2)

def entropie(n1,n2):
    # fonction entropie qui retourne la valeur de l'entropie
    # entres :
        # - n1 : nombre de valeurs dans la classe 0 (Survived=0)
        # - n2 : nombre de valeurs dans la classe 1 (Survived=1)
    # sortie :
        # - valeur de l'entropie
    p1 = n1/(n1+n2)
    p2 = n2/(n1+n2)
    if n1 == 0 or n2 == 0:
        return 0
    else:
        return -p1*log2(p1)-p2*log2(p2)

def gain(S,l):
    # fonction gain d'entropie qui retourne la valeur du gain d'entropie d'un attribut
    # entres :
        # - S : nb observations dans chaque classe
        # - l
    # sorties : 
        # - g : valeur du gain d'entropie
    sp,sn = S
    nb = sp + sn 
    ent_s = entropie(sp,sn)
    g = ent_s
    for ap,an in l:
        g = g - (ap+an)/nb * entropie(ap,an)
    return g

def prediction(label,pred):
    # fonction pour calculer le taux de prdiction sur la base test (prdiction juste)
    # entres :
        # - label : vecteurs des valeurs attendues (Y_test)
        # - pred : vecteurs des prdiction de X_test
    # sorties : 
        # - taux : taux de prdiction (en %)
    sum = 0
    n = len(label)
    for i,j in zip(label,pred): # boucle simultane
        if i == j:
            sum += 1  
    taux = sum/n
    return taux
